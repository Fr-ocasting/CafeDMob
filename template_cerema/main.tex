% Configuration de \heading1 (mappé sur \section)
% Arial, Bold, 14pt (correspond à \large en document 12pt), All Caps, Couleur RGB{41, 37, 116}
% Hanging 0.76cm, Spacing: Before 24pt, After 10pt
\titleformat{\section}
  {\arialfont\bfseries\large\color{ceremaDarkBlue}\MakeUppercase} % Format police
  {\thesection} % Label (Numéro)
  {0.76cm}      % Espace entre le numéro et le titre (Hanging)
  {}
\titlespacing*{\section}{0pt}{24pt}{10pt}

% Configuration de \heading2 (mappé sur \subsection)
% Arial, Bold, 12pt (correspond à \normalsize), Couleur RGB{41, 37, 116}
% Hanging 1.02cm, Spacing: Before 6pt, After 10pt
\titleformat{\subsection}
  {\arialfont\bfseries\normalsize\color{ceremaDarkBlue}}
  {\thesubsection}
  {1.02cm}
  {}
\titlespacing*{\subsection}{0pt}{6pt}{10pt}

% Configuration de \heading3 (mappé sur \subsubsection)
% Arial, Bold, 12pt (correspond à \normalsize), Couleur RGB{41, 37, 116}
% Hanging 1.27cm, Spacing: Before 4pt, After 4pt
\titleformat{\subsubsection}
  {\arialfont\bfseries\normalsize\color{ceremaDarkBlue}}
  {\thesubsubsection}
  {1.27cm}
  {}
\titlespacing*{\subsubsection}{0pt}{4pt}{4pt}
